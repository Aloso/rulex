#! flavor=Pcre
test {
  match '3';
  match '-2+4';
  match '(2+4)';
  match '1+(2+4)+5';
  match '(3+(-4)+-(5/4))*2';

  reject '';
  reject '()';
  reject '+4';
  reject '1-';
  reject '(2+3';
  reject '2+3)';
  reject 'e';
  reject '3 + 4';
}

let number = [ascii_digit]+;
let sign = '-';
let op = ['+-*/'];

sign? ('(' recursion ')' | number) (op recursion)*
-----
-?(?:\(\g<0>\)|[0-9]+)(?:[+\-*/]\g<0>)*
